---creating tables related to eateries and flights..

Note - 1) First create user table as user# is used as a foreign key.  (imp)
       2) then create address table because those addresses are also used. (imp)


create table travel_booking1(
travel# int,
user# int,
country varchar(20),
city varchar(20),
total_price int,
primary key(travel#),
foreign key(user#) references user1
);                        

create table eatery1(
eatery# int,
address# int,
name varchar(100),
cuisine varchar(50),
valet varchar(5),
capacity_avail int,
cost_per_person int,
primary key(eatery#),
foreign key(address#) references address1 
);                          

create table eatery_timings1(
eatery# int,
open_timing varchar(20),
close_timing varchar(20),
foreign key(eatery#) references eatery1
);

create table eatery_booking1(
travel# int,
eatery# int,
booking_date date,
capacity_booked int,
foreign key(travel#) references travel_booking1,
foreign key(eatery#) references eatery1
);

create table eatery_review1(
eatery# int,
user# int,
stars int,
line1 varchar(1000),
line2 varchar(1000),
summary varchar(100),
foreign key(eatery#) references eatery1,
foreign key(user#) references user1
);                                             


create table airline1(
airline# int,
name varchar(20),
check_in_luggage varchar(20),
hand_luggage varchar(20),
primary key(airline#)
);

create table airport1(
airport# varchar(10),
address# int,
primary key(airport#)
);

create table flight1(
flight# int,
airline# int,
takeoff_date date,
takeoff_time varchar(15),
landing_date date,
landing_time varchar(15),
arrival_airport varchar(20),
departure_airport varchar(20),
price int,
primary key(flight#),
foreign key(airline#) references airline1,
foreign key(arrival_airport) references airport1(airport#),
foreign key(departure_airport) references airport1(airport#)
);


create table available_seats1(
flight# int,
seat_no int,
foreign key(flight#) references flight1,
primary key(flight#,seat_no)
);

create table flight_booking1(
travel# int,
flight# int,
seat_no int,
meals varchar(30),
primary key(flight#,seat_no),
foreign key(travel#) references travel_booking1
);
